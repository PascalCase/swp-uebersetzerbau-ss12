<funcs>		::= <func> <funcs> | "@"
<func>		::= "def" <type> "id" "(" <optparams> ")" <block> | "def" <type> "id" "(" <optparams> ")" ";"
<optparams>	::= <params> | "@"
<params>	::= <type> "id" "," <params> | <type> "id"

<decls>		::= <decls> <decl> | "@"
<decl>		::= <type> "id" ";"
<type>		::= <type> "[" "num" "]" | "basic" | "record" "{" <decls> "}"
<stmts>		::= <stmts> <stmt> | "@"
